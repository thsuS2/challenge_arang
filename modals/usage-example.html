모달 팝업 사용 예시

사용 방법

// 1. HTML에 모달 파일 include
<!-- 기록 완료 모달 -->
<div id="recordCompleteModalContainer"></div>

<!-- 포인트 지급 모달 -->
<div id="pointRewardModalContainer"></div>

// 2. JavaScript에서 모달 표시
// 기록 완료 모달
showRecordCompleteModal();

// 포인트 지급 모달 (포인트 지정 가능)
showPointRewardModal(100, 1250); // 획득 포인트, 총 포인트
